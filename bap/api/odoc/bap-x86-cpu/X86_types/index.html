type rex = {
  1. rex_w : bool;
    (*

    Bit 3: 1 = 64-bit operand size

    *)
  2. rex_r : bool;
    (*

    Bit 2: Extension of ModR/M reg field

    *)
  3. rex_x : bool;
    (*

    Bit 1: Extension of SIB index field

    *)
  4. rex_b : bool;
    (*

    Bit 0: Extension of ModR/M r/m field, SIB base field, or opcode reg field

    *)
}
