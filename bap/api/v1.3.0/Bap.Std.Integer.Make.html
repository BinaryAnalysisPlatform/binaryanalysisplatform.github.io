A common set of infix operators


val (~-) : t -> t
~-x = neg x
val (+) : t -> t -> t
x + y = add x y
val (-) : t -> t -> t
x - y = sub x y
val ( * ) : t -> t -> t
x * y = mul x y
val (/) : t -> t -> t
x / y = div x y
val (mod) : t -> t -> t
x mod y = modulo x y
val (land) : t -> t -> t
x land y = logand x y
val (lor) : t -> t -> t
x lor y = logor x y
val (lxor) : t -> t -> t
lxor x y = logxor x y
val (lsl) : t -> t -> t
x lsl y = lshift x y
val (lsr) : t -> t -> t
x lsr y = rshift x y
val (asr) : t -> t -> t
x asr y = arshift x y
