Module Bap.Std.Block

module Block: sig .. end
Basic block of machine instructions.
type t = block 
include Bap.Std.Block_accessors
include Bap.Std.Block_traverse
module Graph: Graph 
      with type node = t
       and type Node.label = t
       and type Edge.label = edge
Graph of blocks. 
val to_graph : ?bound:(addr -> bool) -> t -> Graph.t
to_graph ?bound entry builds a graph starting with entry and spanning all reachable blocks.
bound : if specified, then the resulting graph will contain only blocks b for which bound (Block.addr b) evaluated to true. Symtab.create_bound is an example of such function.
