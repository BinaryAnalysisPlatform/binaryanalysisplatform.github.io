Module type Bap.Std.Node

module type Node = sig .. end
Interface provided by graph nodes.
type t
node type is opaque
type graph 
type label 
type edge 
val create : label -> t
val label : t -> label
label node returns a node's label
val mem : t -> graph -> bool
mem node graph is true if node is part of graph
val succs : t -> graph -> t seq
succs node graph returns a sequence of successors of a node in a given graph
val preds : t -> graph -> t seq
preds node graph returns a sequence of predecessors of a node in a given graph
val inputs : t -> graph -> edge seq
inputs node graph is incomming edges of a node in graph
val outputs : t -> graph -> edge seq
outputs node graph is outcomming edges of a node in graph
val insert : t -> graph -> graph
insert node graph returns new graph that contains node as well as all other nodes of old graph.
val update : t -> graph -> graph
update n g updates node n in a graph g if n was contained in g. Otherwise returns g unchanged. Note: this operation makes sense for nodes, that are not structurally compared, and have some fields, that doesn't affect the equality of a node.
val remove : t -> graph -> graph
remove n g returns a new graph g', that is structurally the same as g but doesn't contain node n
val has_edge : t -> t -> graph -> bool
has_edge x y g is true if there exists such edge in graph g that its destination is y and source is x
val edge : t ->
       t -> graph -> edge option
edge x y g if there is an edge in graph g with source x and destination y, then return it. Otherwise return None
val edges : t ->
       t -> graph -> edge seq
edges x y g returns all edges in graph g that have source node x and destination node y
include Bap.Std.Opaque
node provides common data structures, like Table, Map, Set, Hash_set, etc.
